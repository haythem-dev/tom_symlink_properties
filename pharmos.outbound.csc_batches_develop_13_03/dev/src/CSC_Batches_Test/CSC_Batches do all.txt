-- ******************************************************************************
--
-- 				Drop Tables
--
-- ******************************************************************************
CREATE PROCEDURE drop_table()
  DEFINE c INT;

  set debug file to "SQLDebugLogfile";
  trace on;

  SELECT count(*) into c from systables where lower(tabname)="orderhead";
  if c=1 then
    drop table orderhead;
  end if;

  SELECT count(*) into c from systables where lower(tabname)="orderpos";
  if c=1 then
    drop table orderpos;
  end if;

  SELECT count(*) into c from systables where lower(tabname)="cscini";
  if c=1 then
    drop table cscini;
  end if;

  SELECT count(*) into c from systables where lower(tabname)="csctransaction";
  if c=1 then
    drop table csctransaction;
  end if;

END PROCEDURE;
EXECUTE PROCEDURE drop_table();
DROP PROCEDURE drop_table();

CREATE UNIQUE INDEX XPKOrderHead  ON OrderHead 
(
        BranchNo               ASC,
        OrderNo                ASC
);

-- ******************************************************************************
--
-- 				Create Table CscIni
--
-- This is an ini table for csc programs. The table consists of key/value pairs.
-- Example:
-- If in Branch no 5 there should be visible only ordertype "UE" and "ER" these 
-- the recordsets to be inserted:
-- 	BranchNo=5, Key="ordertype", Value="UE"
--	BranchNo=5, Key="ordertype", Value="ER"
--
--
-- ******************************************************************************

CREATE TABLE CscIni(
BranchNo smallint not null,		-- Branch number
Key char(20) not null,			-- Key of key/value pair
Value char(20) not null			-- Value of key/value pair
);

CREATE UNIQUE INDEX XPKCscIni ON CscIni
(
        BranchNo           ASC,
        Key                ASC,
        Value              ASC
)
;

-- ******************************************************************************
--
-- 				Create Table CscTransaction
--
-- This is a table for realizing transactions in csc batch order processing. 
-- The table rows consist of an order numbers and an application id.
-- Example:
-- If application csc_batch starts a transaction for order no 112, 
-- there is an entry 'csc_batch', 112 in the table.
-- When the transaction is committed, the row is deleted.
--
-- ******************************************************************************

CREATE TABLE CscTransaction (
BranchNo smallint not null,		-- Branch number
OrderNo int not null,		-- Order number
PosNo int not null,		-- Position number
Application char(20) not null	-- Name of Application
);

CREATE UNIQUE INDEX XPKCscTransaction ON CscTransaction
(
        BranchNo           ASC,
        OrderNo            ASC,
        PosNo              ASC,
        Application        ASC
)
;
